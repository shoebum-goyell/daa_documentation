void splitPoints(List<Custom_Point> points, Line medianLine, List<Custom_Point> tLeft, List<Custom_Point> tRight) {
  double medianX = medianLine.p1.x;
  for (Custom_Point point in points) {
    if (point.x < medianX) {
      tLeft.add(point);
    } else if (point.x > medianX) {
      tRight.add(point);
    } else {
      tLeft.add(point);
    }
  }
}
